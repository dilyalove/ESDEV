function validate() {
	let userName = document.forms['form']['userName'].value;
	let userEmail = document.forms['form']['userEmail'].value;

	if (userName.length==0) {
		document.getElementById('errorName').innerHTML = 'Введите, пожалуйста, имя.';
		return false;
	}

	if (userEmail.length==0) {
		document.getElementById('errorEmail').innerHTML = 'Введите, пожалуйста, email.';
		return false;
	}

	at=userEmail('@');
	dot=userEmail('.');
	if (at<1 || dot <1) {
		document.getElementById('errorEmail').innerHTML = 'Некорректный email.';
		return false;
	}
};
